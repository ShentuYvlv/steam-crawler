import asyncio
import httpx
from src.utils.oxylabs_proxy import load_oxylabs_proxy_settings

settings = load_oxylabs_proxy_settings()
print("enabled =", settings.enabled)
print("host =", settings.host)
print("scheme =", settings.scheme)
print("direct_fallback =", settings.direct_fallback)

async def test(label, proxy, url):
    print(f"\n== {label} ==")
    try:
        async with httpx.AsyncClient(
            proxy=proxy,
            timeout=httpx.Timeout(20.0, connect=20.0),
            verify=False,
            follow_redirects=True,
            trust_env=False,
        ) as client:
            r = await client.get(url)
            print("status =", r.status_code)
            print("final_url =", str(r.url))
            print("server =", r.headers.get("server"))
    except Exception as e:
        print(type(e).__name__, str(e))

async def main():
    if not settings.enabled:
        print("Oxylabs is disabled")
        return

    urls = [
        ("steamcommunity", "https://steamcommunity.com/"),
        ("store", "https://store.steampowered.com/"),
        ("oxylabs-location", "https://ip.oxylabs.io/location"),
    ]

    # rotating
    for label, url in urls:
        for scheme in ("https", "http"):
            proxy = settings.build_rotating_proxy_url(scheme_override=scheme)
            await test(f"rotating {scheme} {label}", proxy, url)

    # sticky
    sticky_port = settings.choose_sticky_session_port()
    for label, url in urls:
        for scheme in ("https", "http"):
            proxy = settings.build_sticky_proxy_url(session_port=sticky_port, scheme_override=scheme)
            await test(f"sticky {scheme} {label}", proxy, url)

asyncio.run(main())
